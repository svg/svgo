        <stop offset="0" stop-color="#ff0000"/>
        <stop offset="1" stop-color="#0000ff"/>

    <path fill="url(#gradient)" d="M 0 0 H 100 V 80 H 0 z"/>

    <path style="fill:url(#gradient)" d="M 200 100 H 300 v 80 H 200 z"/>
