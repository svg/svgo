Removes <script> elements even when script is declared with an explicit
SVG or XHTML namespace.

===

<svg xmlns="http://www.w3.org/2000/svg" xmlns:svg="http://www.w3.org/2000/svg" xmlns:xhtml="http://www.w3.org/1999/xhtml" viewBox="0 0 100 100" version="1.1">
  <script>
    alert('This alert is from the default namespace.');
  </script>
  <svg:script>
    alert('This alert is from the explicit/prefixed SVG namespace.');
  </svg:script>
  <xhtml:script>
    alert('This alert is from the explicit/prefixed XHTML namespace.');
  </xhtml:script>
</svg>
